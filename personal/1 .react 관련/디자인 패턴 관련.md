```tsx
const SelectInternal: FC<{ children?: ReactNode }> = ({ children }) => {
  return <select>{children}</select>;
};

const Option: FC<{
  value?: string;
  children?: ReactNode;
}> = ({ value, children }) => {
  return <option value={value}>{children}</option>;
};

const Select = Object.assign(SelectInternal, { Option });

export { Select };
```
