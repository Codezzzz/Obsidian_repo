from heapq import heappush, heappop, heapify
def solution(A, B):
    answer = 0

    heapify(A)
    heapify(B)
    while A:
        a = heappop(A)
        while B:
            b = heappop(B)

            if b > a:
                answer += 1
                break
    return answer
